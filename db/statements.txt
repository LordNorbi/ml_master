UPDATE Taskset
SET Exit_value = -2
WHERE TaskSet.Set_ID in (select TaskSet.Set_ID
from TaskSet
inner join Job on TaskSet.Set_ID = Job.Set_ID
where Job.Exit_Value = 'EXIT_PERIOD')
